                   00000   - SUCCESSFUL EXECTUTION OF SQL STATEMENT
                   02000   - NOT FOUND
                   01???   - WARNING

                       SELECT;
                         WHEN SQLState = '00000';
                           SQLSUCCESS = *ON;
                         WHEN SQLState = '02000';
                           NOTFOUND = *ON;
                         WHEN %Subst(SQLState : 1 : 2 ) = '01';

                         OTHER;

                       ENDSL;
